copyText(button, location.href, function (text) {
    console.log('复制成功：' + text);
});
copyText(button2, output, function (text) {
    console.log('复制成功：' + text);
});

button3.onclick = function () {
    copyText('CSS新世界', function (text) {
        console.log('复制成功：' + text);
    });
    // 内置的复制成功提示
    copyText.tips(event);
    // 改变按钮状态
    this.setAttribute('type', 'success');
};
